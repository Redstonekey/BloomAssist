def soil_moisture_level(sensor_value):
    if sensor_value >= 1023:
        return "Sehr Trocken"
    elif 900 <= sensor_value < 1023:
        return "Trocken"
    elif 750 <= sensor_value < 900:
        return "Feucht"
    elif 634 <= sensor_value < 750:
        return "Sehr feucht"
    elif sensor_value <= 633:
        return "Nass"
    else:
        return "ERROR 301"
